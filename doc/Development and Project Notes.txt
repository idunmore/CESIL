@staticmethod
def is_legal_identifier(identifier):
        # Identifiers must be between 1 and 6 characters, start with an alpha
        # and contain only alpha-numerics
        legality = False
        if not CESIL.is_instruction(identifier):
            if len(identifier) > 0 or len(identifier) <= 6: 
                if identifier[0].isalpha():
                    if len(identifier) == 1 or identifier[1:].isalnum():
                        legality = True
        return legality
